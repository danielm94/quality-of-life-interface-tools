# Corresponds to global_edicts_commander_events.3.b (Moved logic from original event 4)
country_event = {
	id = global_edict_commander_option_events.6
	hidden = yes
	is_triggered_only = yes
	immediate = {
		every_owned_province = {
			limit = {
				is_state = yes
				can_change_state_edict = yes
				NOT = {
					has_state_edict = edict_defensive_edict
				}
				any_province_in_state = {
					has_fort_building_trigger = yes
				}
			}
			save_event_target_as = province_to_check
			event_target:province_to_check = {
				if = {
					limit = {
						has_siege = yes
					}
					edict_commander_set_edict_for_state = {
						edict = edict_defensive_edict
					}
				}
				else_if = {
					limit = {
						any_neighbor_province = {
							has_rebels_in_province = yes
						}
					}
					edict_commander_set_edict_for_state = {
						edict = edict_defensive_edict
					}
				}
				else_if = {
					limit = {
						owner = {
							is_at_war = yes
						}
					}
					owner = {
						every_war_enemy_country = {
							save_event_target_as = war_enemy
							event_target:province_to_check = {
								if = {
									limit = {
										any_neighbor_province = {
											num_of_units_in_province = {
												amount = 1
												who = event_target:war_enemy
											}
										}
									}
									edict_commander_set_edict_for_state = {
										edict = edict_defensive_edict
									}
								}
							}
						}
					}
				}
			}
		}
	}
	option = {
		name = "OK"
	}
	# Dummy option
}
