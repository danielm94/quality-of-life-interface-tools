custom_button = {
	name = open_global_edict_button
	tooltip = open_global_edict_button_tt
	potential = {
		is_state = yes
		owned_by = FROM
	}
	trigger = {
	}
	effect = {
		FROM = {
			if = {
				limit = {
					FROM = {
						NOT = {
							has_country_flag = global_edicts_window_open
						}
					}
				}
				set_country_flag = global_edicts_window_open
			}
			else = {
				clr_country_flag = global_edicts_window_open
			}
		}
	}
}
