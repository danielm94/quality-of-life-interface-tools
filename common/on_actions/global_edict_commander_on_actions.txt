on_startup = {
	edict_commander_set_all_trade_node_flags = yes
}

on_province_religion_converted = {
	if = {
		limit = {
			is_state = yes
			has_state_edict = edict_religious_unity
			religion = owner
		}
		owner = {
			save_event_target_as = dude
		}
		set_variable = {
			which = remove_the_edict
			value = 1
		}
		every_province_in_state = {
			limit = {
				owned_by = event_target:dude
			}
			if = {
				limit = {
					NOT = {
						religion = event_target:dude
					}
				}
				set_variable = {
					which = remove_the_edict
					value = 0
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = remove_the_edict
					value = 1
				}
			}
			set_state_edict = no_edict
		}
	}
}
