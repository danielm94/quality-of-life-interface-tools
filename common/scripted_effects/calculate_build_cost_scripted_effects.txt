#SCOPE = COUNTRY
calculate_building_action_cost = {
	set_variable = {
		which = base_cost
		value = $base_cost$
	}
	set_variable = {
		which = $var_name$
		which = base_cost
	}
	multiply_variable = {
		which = $var_name$
		which = $build_cost_var$
	}
	change_variable = {
		which = $var_name$
		which = base_cost
	}
}
#SCOPE = COUNTRY
calculate_trade_depot_upgrade_cost = {
	calculate_building_action_cost = {
		base_cost = 200
		var_name = trade_depot_upgrade_cost
		build_cost_var = country_build_cost

}
